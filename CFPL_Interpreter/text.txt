VAR x AS INT
VAR num, count=0 AS INT
START
    num = 10
    x = 0
    WHILE (num > 5)
        START
            num = num - 1
        STOP
    IF (num == 5)
        START
            x = num
            IF (x >= 5)
                START
                    WHILE (count < num)
                        START
                            x = x + 1
                            count = count + 2
                        STOP
                STOP
            ELSE
                START
                    x = x / num
                STOP
        STOP
    OUTPUT: "num = " & num & ", x = " & x & ", count = " & count
STOP
